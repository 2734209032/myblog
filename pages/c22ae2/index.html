// 用promise封装ajax请求
// fetch(url).then(res => {})
// // 我们自己约定，这个是get请求
// ajax(url).then(res => {})

// // 这种是post请求的方式
// ajax(url, params).then(res => {

// })

// 1. 明确，ajax()调用之后的返回值是一个promise对象 
const ajax = (url, params) => {
  return new Promise((resolve, reject) => {
    // 发送ajax请求 
    const xhr = new XMLHttpRequest()

    // 2.1 我们自己约定，不传params，表示get请求
    if (!params) {
      xhr.open('GET', url)
      xhr.send()
      // 2.2 如果传了params，表示post请求
    } else if (params) {
      xhr.open('POST', url)
      xhr.setRequestHeader('Content-type', 'application/json')
      // params我们写参数的时候，是一个对象
      xhr.send(JSON.stringify(params))
    }
    // 3. 监听状态
    xhr.onreadystatechange = function () {
      if (xhr.readyState !== 4) return
      if (xhr.status >= 200 && xhr.status < 300 || xhr.status === 304) {
        //xhr.response
        // 当调用了resovle这个方法之后，可以执行 .then(res => {}) , res就是resolve传进去的值 
        resolve(JSON.parse(xhr.response))
      } else {
        // .catch(err => {})  err ==> xhr.staus
        reject(xhr.status)
      }
    }
    // 4.监听error
    xhr.onerror = function () {
      reject('你的网络开小差了')
    }

  })
}

02.readyState和status

 // 1. xhr.readyState

    // readyState ==> 表示的是xhr这个对象当前所处的一个状态，总是下列的某一种情况

    // 0 UNSENT 代理被创建，但尚未调用 open() 方法。
    // 1 OPENED open() 方法已经被调用。
    // 2 HEADERS_RECEIVED send() 方法已经被调用，并且头部和状态已经可获得。
    // 3 LOADING 下载中；responseText 属性已经包含部分数据。
    // 4 DONE 下载操作已完成。


    // 2. xhr.status  ===> http状态码
    // ==> 请求成功的状态码
    // xhr.status >= 200 && xhr.status < 300 || xhr.status === 304

    const api_url = 'http://192.168.19.66:8888/api/students'
    // 1. 创建xhr对象
    const xhr = new XMLHttpRequest()
    console.log('0 - UNSENT', xhr.readyState)

    // 2. 设置请求方法和接口地址
    xhr.open('GET', api_url)
    console.log('1 - OPENED', xhr.readyState)

    // 3. 发送请求
    xhr.send()

    // 4. 监听readyState的状态
    // dom-0的方式监听的  ， 监听的就是readystate
    xhr.onreadystatechange = function () {
      // 2. ===> 表示send方法已经被调用
      // 3. ===> 后端给我们前端的数据正在下载中
      // 4. ===> 数据下载完成
      console.log(xhr.readyState)
    }

# 8.promise的静态方法，all和race

<script>
    function timePromise(t) {
      return new Promise((resolve, reject) => {
        setTimeout(() => {
          //   console.log(1)
          resolve(t / 1000)
        }, t)
      })
    }
    const p1 = timePromise(1000)
    const p2 = timePromise(2000)
    const p3 = timePromise(3000)
    const p4 = timePromise(4000)
    const p5 = new Promise((resolve, reject) => {
      setTimeout(() => {
        //   console.log(1)
        // resolve(t / 1000)
        reject('失败了哦，仙人')
      }, 1000)
    })
    //Promise.all里面传的是一个数组（数组的每一项都是一个promise）,
    //只有每一个promise都成功了，才会走进.then,有一个失败了就会走进.catch
    Promise.all([p1, p2, p3, p4]).then(res => {         //静态方法
      console.log(res)

    }).catch(err => {
      console.log(err)
    })
    console.dir(Promise)
    console.dir(new Promise(() => { }))
    // ------------------------race-------------------
    // Promise.race里面传的也是一个数组（数组里面的每一项都是一个promsie），
    // 等待最先成功的那一个，只要有一个成功就会走进.then
    const res = Promise.race([p1, p2, p3, p4]).then(res => {
      console.log(1111, res)

    })
    console.log(res)

  </script>

 function myAxios(options) {
    const { url, method = 'get', data } = options
    return new Promise((resolve, reject) => {
      // 1.实例化异步对象
      const xhr = new XMLHttpRequest()
      // 2.设置请求的 地址 和方法
      xhr.open(method, url)
      xhr.setRequestHeader('content-type', 'application/json')
      // 3.注册回调函数 服务器响应内容回来之后触发
      xhr.addEventListener('load', function () {
        console.log(xhr.response)
        resolve(JSON.parse(xhr.response))
      })
      // 4.发送请求
      xhr.send(JSON.stringify(data))
    })
  }
  const p1 = myAxios({
    url: 'http://ajax-api.itheima.net/api/news',
  })
  const p2 = myAxios({
    url: 'http://ajax-api.itheima.net/api/books',
  })
  Promise.all([p1, p2]).then((res) => {
    console.log(res)
    // 渲染页面
  })
  Promise.race([p1, p2]).then((res) => {
    console.log('race', res)
  })
