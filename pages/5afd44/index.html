# 1.获取dom useRef

import { useRef, useState } from "react";
const InputRef = () => {
  const inputRef = useRef(null);
  const [value, setValue] = useState("");

  return (
    <div>
      <input  onInput={(e)=>setValue(e.target.value)}/>
      <button onClick={() => console.log(value)}>
        获取第一个input的值
      </button>
      <input ref={inputRef} />
      <button onClick={() =>        console.log(inputRef.current?.value)}>
    //?代表可选链，严谨性
        获取第二个input的值
      </button>
    </div>
  );
};
export default InputRef;
//注意点
1.根据业务进行 DOM 操作：通过 inputRef.current 拿到 DOM 对象
2.不要在组件渲染时使用 ref 进行 DOM 操作，因此，此时 ref 还没有值
3.操作文本框的值，推荐使用状态（受控组件）操作

# 2.组件传值

1.通过props进行接收，绑定属性的方式
//父组件
import Children from "./computend/Children";
import dataList from "./data.json";
function App() {
  return (
    <>
      {dataList.house.map((item, index) => {
      //通过绑定属性的方式个子组件传值
        return <Children key={index} item={item} />;
      })}
    </>
  );
}
//子组件中
import './children.css'
//默认是props，可以解构的，我是解构了的
const Children = ({item}) => {
  return <>
  <div className="box">
    <div className="time">{item.ptime}</div>
    <div className="title">{item.title}</div>
    <div className="user">{item.source}</div>
    <div className="usrinfo">{item.category}</div>
  </div>
  
  </>
}
export default Children;
