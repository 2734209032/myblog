node起服务器代理项目

# 1.静态文件代理

//cp.js
const http = require('http')
// 操作文件
const { createReadStream } = require('fs')
const mime = require('mime')  //需要安装   cnpm i mime

const server = http.createServer((req, res) => {
    console.log(req.url)
    res.statusCode = 200
    // 如果你要的 path /test/test 没有 .xxx 结尾
    if (req.url.split('.').length <= 1) {
        req.url = '/index.html'
    }
    // createReadStream 创建读文件流
    const rs = createReadStream('./inquiry/dist' + req.url) //是我们项目打包后的地址
    res.setHeader('content-type', mime.getType(req.url))
    // 文件读出来，丢给 res
    rs.pipe(res)
    rs.on('end', () => {
        res.end()
    })
})
server.listen(3001, '0.0.0.0', () => {
    const addr = server.address()
    console.log(`server's running at ${addr.address}:${addr.port}`)
})
//注意，cp.js不能在我们项目里面 ， 与项目同级，不然项目用的是es6模块，而我们用的是conmm.js模块，运行 node ./cp.js
//访问，http://localhost:3001 或 http://192.168.19.47:3001/home  查看ip地址 ipconfig
