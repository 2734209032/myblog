# 1.下载i18n插件

cnpm i vue-i18n@7.3.3  //版本根据情况而定，我当时下载的最高版本，结果Vue.use注册的时候报错插件没有install方法

# 2.引入i18n，注册插件

import VueI18n from 'vue-i18n'
// console.dir(VueI18n)
Vue.use(VueI18n)

# 3.创建中英映射的语言包

// 1、创建中文语言包对象
const zh = {
  username: '用户名',
  email: '邮箱',
  mobile: '手机'
}
// 2、创建英文语言包对象
const en = {
  username: 'Username',
  email: 'Email',
  mobile: 'Mobile'
}
//实际项目不可能只有这么点需要实习国际化的，我们一般会在src创建一个lang文件夹，中文的语言包定义一个zh.js，英文的定义一个en.js，最后引入到main.js

# 4.创建Vuei18n实例

// 3、组合语言包对象
const messages = {
  zh,
  en
}
console.log(navigator.language)//我这里它给我获取的是zh-CN,直接给我转化为英文小写，然后就用了下面的截取方法
// 4、创建 VueI18n 实例，并为 messages 和 locale 属性赋值
const i18n = new VueI18n({
  messages,
  locale: navigator.language.toLowerCase().split('-')[0],// 获取浏览器的语言
  silentTranslationWarn: true,
})
//挂载到vue上
new Vue({
  i18n,
  render: h => h(App),
}).$mount('#app')

# 5.在模版中使用

<template>
      <div>
        <!-- 国际化 -->   
         <button @click="translate('zh')">切换为中文</button>
          <button @click="translate('en')">切换为英文</button>
          <div>{{ $t('username') }}</div>
           <div>{{ $t('email') }}</div>
            <div>{{ $t('mobile') }}</div>
            </div>
            </template>
            <script>​
        export default{
              methods:{
                    translate(lang) {
                          this.$i18n.locale = lang 
                        }, 
                      }}
                      </script>
          <style></style>
          //如果要搭配elementUI使用，它也支持i18n可以去看看，我现在只是了解i18n的使用
