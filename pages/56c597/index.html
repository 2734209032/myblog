# 2.pinia持久化

1.安装 cnpm i pinia-plugin-persistedstate
2.// 引入pinia持久化插件，自动存储到本地储存
import presist from 'pinia-plugin-persistedstate'
const app = createApp(App)
app.use(createPinia().use(presist)) //必须在user pinia的后面use插件
3.配置pinia
import {defineStore} from 'pinia'
const useStore = defineStore(
'cp-user', //id
() => {
  //user 自定义组合函数 里面定义state actions getters
},
{
  persist: true, //开启pinia持久化
}
)
export default useStore

# 5.移动端使用vant组件的tabbar组件与路由联动

//概念：点击切换路由，在网页输入路由，相应的tabbar高亮
<script setup lang="ts">
import { useRoute, useRouter } from "vue-router";
import { useUserStore } from "./stores";
import Login from "./views/Login.vue";
import { Tabbar, TabbarItem } from "vant";
const log = () => {
  console.log("click");
};
const userStore = useUserStore();
const route = useRoute();
const router = useRouter();
</script>

<template>
  <div v-if="userStore.user?.token">
    <!-- 有 token 才能访问内容 -->
    <!-- 路由只有在真正登录以后，才开始起作用 -->
    <router-view />
    <Tabbar
      v-if="route.path === '/home' || route.path === '/mine'"
      :modelValue="route.path"
      @update:modelValue="(path) => router.push(path)"
    >
      <TabbarItem icon="home-o" name="/home">首页</TabbarItem>
      <TabbarItem icon="search" name="/mine">我的</TabbarItem>
    </Tabbar>
  </div>

  <template v-else>
    <Login />
  </template>
</template>
