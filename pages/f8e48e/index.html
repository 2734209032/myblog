# 2.基础hook与vue3api对应关系

1.useState  ==>   ref
 //  useState 返回了一个数组，第一个是状态，第二个是修改状态的函数
用法:   const [count, setCount] = useState({ value: 0 });
2.useEffect / useLayoutEffect  ==> watch / watch(pre模式)
用法:/// useEffect 函数中，返回的函数，就是 vue中的onCleanup
useEffect(() => { return () => {} }, [a])
3.useContext ==> inject
4.useCallback ==> computed //缓存函数
5.useMemo  ===> computed
//用法   const ab = useMemo(() => a + b, [a, b]);
6.useRef  ==> let 变量 //因为，每次数据发生改变时，react会重新直接调用组件函数，导致变量重新声明，这时这个hook会缓存变量
  const data = useRef(0);
  console.log(data.current) //需要用current取值

function App() {
  console.log("render");
  // useState 返回了一个数组，第一个是状态，第二个是修改状态的函数
  // 第一个 api useState / vue 的 ref
  // count 每次 App render 重新跑的时候，值是新值，引用是不是新引用
  // **纯变更**  count 在一次渲染中，保持**不变性**
  const [count, setCount] = useState({ value: 0 });
  return (
    <div>
      <div>{count.value}</div>
      {/* 小驼峰命名事件 */}
      <button
        onClick={() => {
          // 数组：
          // vue 用修改原数组：push pop shift unshift splice sort reverse
          // react 用不修改原数组：concat slice filter map reduce
          // 对象实现纯变更：深浅拷贝
          // 不可变数据结构：immutable.js immer.js
          // 任何属性的修改，沿着引用路径修改路径上的引用
          // draft 草稿
          setCount(
            produce((draft) => {
              draft.value++;
            })
          );
        }}
      >
        +
      </button>
    </div>
  );
}
