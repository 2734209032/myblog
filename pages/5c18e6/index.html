import qqmapsdk from '../../libs/qqmap-wx-jssdk/index.js'  
//跟以前一样的
  export default {
    name: '',
    components: {},
    data() {
      return {
        longitude: 103.990482,
        latitude: 30.720042,
        start: {
          longitude: 103.996184,
          latitude: 30.721659,
          width:25,
          height:25,
          //#ifdef H5 || MP-WEIXIN     条件编译，因为我们app端的和h5和小程序不一样，app端改不了图片的大小，所以只有引入小的图片
          iconPath:'/static/start.png',
          //#endif
          //#ifdef APP-PLUS
          iconPath:'/static/app-plus/start.png',
          //#endif
        },
        end: {
          longitude: 103.984899,
          latitude: 30.719003,
          width:25,
          height:25,
          //#ifdef H5 || MP-WEIXIN
          iconPath:'/static/end.png',
          //#endif
          //#ifdef APP-PLUS
          iconPath:'/static/app-plus/end.png',
          //#endif
        },
        polyline:[]
      };
    },
    created() {},
    mounted() {},
    onLoad() {
       qqmapsdk.direction({
            mode: 'driving',//可选值：'driving'（驾车）、'walking'（步行）、'bicycling'（骑行），不填默认：'driving',可不填
            //from参数不填默认当前地址
            from:this.start,
            to: this.end, 
            success: (res)=> {  //改箭头函数，方便下面赋值
              console.log(res);
              var ret = res;
              var coors = ret.result.routes[0].polyline, pl = [];
              //坐标解压（返回的点串坐标，通过前向差分进行压缩）
              var kr = 1000000;
              for (var i = 2; i < coors.length; i++) {
                coors[i] = Number(coors[i - 2]) + Number(coors[i]) / kr;
              }
              //将解压后的坐标放入点串数组pl中
              for (var i = 0; i < coors.length; i += 2) {
                pl.push({ latitude: coors[i], longitude: coors[i + 1] })
              }
              console.log(pl)
              //设置polyline属性，将路线显示出来,将解压坐标第一个数据作为起点 polyline
               this.latitude = pl[0].latitude,
                this.longitude = pl[0].longitude,
                this.polyline =  [{
                  points: pl,
                  color: '#FF0000DD',
                  width: 4
                }]
            
            },
            fail: function (error) {
              console.error(error);
            },
            complete: function (res) {
              console.log(res);
            }
          })
    },
