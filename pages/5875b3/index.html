# 1.pinia的使用

//安装
1. npm i pinia 
2.在main.ts中
import { createPinia } from 'pinia'
 app.use(createPinia())
3.定义一个store //一般我们定义一个store 是以use 开头 eag : useStore
import { defineStore } from 'pinia'
// useStore 可以是 useUser、useCart 之类的任何东西
// 第一个参数是应用程序中 store 的唯一 id
export const useStore = defineStore('main', {
  // other options...
  state() {
    return {
      // state
      count :0
    }
  },
  actions:{
    // actions
    increment(){
      this.count++
    }
  },
  getters:{}
})
4.使用
import { useStore } from '@/store/useStore'
const store = useStore() //不能结构
<div>  pinia
  {{ store.count }}
</div>

# 2.依赖注入

//父组件
// symbol 绝对不会有冲突的，独一无二的值
export const useCountKey: InjectionKey<ReturnType<typeof useCount>> = Symbol(); // 定义一个 symbol
  上面的类型解释
   <typeof useCount> //获取函数useCount的整体函数类型
   <ReturnType<typeof useCount> //获取函数返回值类型
    InjectionKey<ReturnType<typeof useCount>> //获取Inject key类型
// 自定义组合函数
export function useCount() {
  // 定义一个自定义组合函数
  const count = ref(0); // 定义一个响应式的 count
  // +1
  const increment = () => {
    // 定义一个 +1 的方法
    count.value++; // 修改 count
  };
  return {
    count, // 返回响应式的 count
    increment, // 返回 +1 的方法
  };
}
privide('useCountKey',useCount)

//子组件使用
import { useCountKey } from './useStore'
// 为什么有空值？ts 无法确定，inject 之前有无 provide
const store = inject(useCountKey)! // 非空断言
// 给了默认值，如果没有找到 provide，就用默认值
const store2 = inject(useCountKey, {
    count: ref(0),
    increment: () => { }
})
js 下使用：
1. privide(key, store) 在父组件声明 store
2. inject(key) 在子孙组件注入 store
3. 两者的 key 必须保持一致
4. 父组件没有 provide，子孙组件不能 inject
ts 下使用：
1. key 用 symbol 声明
2. key 要用 InjectionKey<Store 的类型>
3. 自定义组合函数，该用 InjectionKey<ReturnType<typeof 自定义组合函数>>
4. inject 时，加上非空断言（如果确定父组件有 provide 的情况）
5. inject 可以传入第二个值，作为默认值，如果没有找到 provide，直接使用默认值

# 4.vue3 + ts 移动端适配

安装插件 postcss-px-to-viewport
cnpm  i postcss-px-to-viewport --save-dev
//vite.config.ts
import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'
import viewPort from 'postcss-px-to-viewport'
// https://vitejs.dev/config/
export default defineConfig({
  plugins: [vue()],
  css:{
    postcss:{
      plugins:[
        viewPort({
          viewportWidth: 375,
        }

  )]
  }}
})
