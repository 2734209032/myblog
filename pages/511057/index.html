# 1.v-model语法糖

 <!-- 
      v-model 是一个语法糖
      作用：=> 将data中的数据和表单元素做一个双向绑定
      原理：针对不同的表单元素（或者说表单元素type的不同属性值），vue底层会让
        v-model实现不同的属性绑定和事件监听 

      1. type="text"
      
      1.1 :value + @input 

        <input type="text" v-model="msg">等价于
        <input type="text" :value="msg" @input="handleInput">

      1.2 :value + @change   ==> v-model.lazy 相当于是触发回车或者失去焦点的时候触发 => @change
        <input type="text" v-model.laze="msg">等价于
        <input type="text" :value="msg" @change="handleInput">

      2. type="checkbox"   ==> :checked + @change 

        <input type="checkbox" v-model="flag">等价于
        <input type="checkbox" :checked="flag" @change="handleInput">
     -->

    //  2.this.$emit('自定义事件名')要写成input
  }
}

# 4.ref 和$refs获取Dom元素 和组件

template>
  <div>
    <h3 ref="myH3">大标题</h3>       /// 给我们要获取的dom元素加上 ref属性
    <p ref="p">这是一个p标签</p>
    <button @click="handleGet">点击获取dom元素</button>
    <!-- 组件 -->
    <my-star ref="star"></my-star>   /// 给我们要获取的组件加上 ref属性

    <button @click="handleComp">点击获取my-star组件</button>
  </div>
</template>

<script>
import MyStar from './components/MyStar.vue'
export default {
  components: {
    MyStar
  },
  methods: {
    handleGet() {
      // 点击的时候，获取h3,p标签，修改他们的文本颜色
      // $event / $emit / $refs  Vue内置
      console.log(this.$refs)
      this.$refs.myH3.style.color = 'orange'
      this.$refs.p.style.color = 'skyblue'
    },
    handleComp() {
      // 通过$refs获取到子组件之后，可以调用子组件内部的方法
      console.log(this.$refs.star)
      this.$refs.star.dance()
      this.$refs.star.sing()

      console.log(this.$refs.star.msg)

      // 父元素可以操作子元素的内容，但是也是不太建议 ，自己的数据自己操作~~
      this.$refs.star.age++

    }

    // 1. 给目标元素添加 ref 属性
    // 2. 通过 this.$refs.xxx 来获取 dom元素

    // 区别： 
    //  1. document.querySelector() ==> 是从这个文档去找第一个这个dom元素
    //  2. ref和 $refs 是获取的当前组件的元素  ，如果有多个相同的ref属性的dom元素，后面的会覆盖前面的
  }
}
</script>

// 所以，不能数据一变化就更新DOM元素（渲染页面内容） ==> 实际情况是先攒着~~，最后一次性的批量更新
methods：{
    handleDom(){
        this.count++

      // 回调函数 ==> 当dom元素更新后，就会立刻执行 
      this.$nextTick(() => {
        console.log(this.$refs.p.innerHTML)

}

全局注册自定义指令

 <p v-color="color1"> 我是普通文本 -橘色</p>
  <h2 v-color="color2">我是标题 -天空蓝</h2>
  </div>
</template>
<script>
export default {
  data() {
    return {
      color1: 'orange',
      color2: 'skyblue'
    }
  },

//在main.js里注册
// 全局注册，放到main.js中，整个组件都可以使用这个指令 
// 局部注册，在某个组件内部注册 ，组件内部使用
Vue.directive('border', {
  // elment => 给哪个标签写上指令，就是获取哪个标签
  // inserted => 当写上指令的那个标签插入到dom元素中，渲染好了，就会执行
  inserted(el) {
    el.style.border = "1px solid skyblue"
    el.style.boxShadow = "0 15px 30px rgb(0 0 0 / 10%)"
  }
})
Vue.directive('color', {
  inserted(el, binding) {
    // (el , {value}) =>对象的解构
    el.style.color = binding.value
  },
  // v-color="xxx" ==> binding.value 
  // update指令的值改变时触发, binding.value的值修改触发   局部注册指令的时候也可以用
  update(el, binding) {
    el.style.color = binding.value
  }
})
