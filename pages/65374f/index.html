# 2.state状态

// 创建仓库 store
const store = new Vuex.Store({
  // state 状态, 即数据, 类似于vue组件中的data,
  // 区别在于 data 是组件自己的数据, 而 state 中的数据整个vue项目的组件都能访问到
  state: {
    count: 101
  }
})

import {mapState} from 'vuex'

}
}

# 3.mutations

const store  = new Vuex.Store({
  state: {
    count: 0
  },
  // 定义mutations  mutations是一个对象，对象中存放修改state的方法
  mutations: { 
// mutation:存放操作数据的方法
// 更改 Vuex 的 store 中的状态的唯一方法是提交 mutation！！！
// 所有的mutation函数，第一个参数都是state
 // 1. 第一个参数，state
    // 2. 第二个参数，payload 载荷，相当于就是我们传递的数据
    // payload可以是任意类型，大多情况下是一个对象
    // mutation必须是同步的！！why？Vuex要求它是同步的！，保证状态的可跟踪性
    // 注意，mutation最多只能有两个参数，如果想传多个参数过来，包装进payload中
      addOne(state,payload){
          state.count+=payload
      }
  },
   // 开启严格模式，必须按着规范来，对于不规范的Vuex用法，会报错
  // 默认是不开启的，实际开发也不需要开启，这个会耗一些性能
  strict: true
})

使用辅助函数

//在组件中 
import {mapMutations} from 'vuex'
export defalut {
    methods:{
        ...mapMutations(['addOne'])   //这个事件名和store/index.js中的mapMutation函数的名字一致，可以写多个
    }
}
//如何调用 ，
1.在组件模板中注册事件，如果事件名和mapMutation函数一致的话，就直接可以调用，传参直接在模板写,@click="addOne(2)"
2.如果不一致的话，在方法中
methods:{
 ...mapMutations(['addOne']) ,  //这里也是需要引入mapMutations
fn(val){
    this.addOne(val)
}}

mutations函数必须是同步的

// 为什么mutation必须是同步的  保证数据状态的可跟踪性
// 在Vue.js和其他一些前端JavaScript框架中，为什么mutation必须是同步的呢?
// 这是因为Vue.js采用了一种基于数据驱动的设计模式，其中状态(state)存储在Vuex存储store)中。为了使Vuex存储在任何时候都保持一致性，必须保证状态的更改是可预测的
// 因此，Vuex要求mutation是同步的，这样才能保证状态的变更在任何时候都是可预测的。异步操作可能会导致状态的更改不可预测，并导致问题。
// 同时，异步操作通常会导致更多的复杂性，并且Vuex中的操作必须是可追踪的，以便调试和跟踪状态的变化。同步操作可以更容易地进行跟踪和调试，因此强制同步是有意义的。

# 4.actions函数处理异步

const state = new Vuex.Store({
  state: {
    count: 100,
    list: [1, 2, 4, 5, 6, 4]
  },mutations: {
    addNum1 (state, payload) {
      state.count += payload
    }},
  actions: {
    // 如果是异步的操作，放到action中处理
    // 1. action可以包含任意的异步操作
    // 2. action中不能修改state数据
    waitToAdd (ctx, payload) {
      setTimeout(() => {
        ctx.commit('addNum1', payload)
      }, 1000)
    }
  },
})

# 5.getters计算属性

const state = new Vuex.Store({
  state: {
    count: 100,
    list: [1, 2, 4, 5, 6, 4]
  },
  getters: {
    // getter相当于是基于state的计算属性
    twoCount (state) {
      return state.count * 2
    },
    listLength (state) {
      return state.list.length
    },
    otherCout (state, getters) {
      // getter计算方法还可以接收2个参数
      return getters.twoCount + getters.listLength //getters.listLength是我们上面定义的getter函数
    }
  },
})

提交模块中的mutation

全局的:   this.$store.commit('mutation函数名', 参数)
模块中的: this.$store.commit('模块名/mutation函数名', 参数)

namespaced: true 后, 要添加映射, 可以加上模块名, 找对应模块的 state/mutations/actions/getters
// 子模块获取state数据
// 1. $store.state.模块名.xxx
// 2. ...mapState('模块名',['xxx','yyy'])  ==>
//   ==> 如果要以mapState映射的方式获取，一定要记得给模块加namespaced命名空间
computed: {

  // 模块中的
  ...mapState('user', ['myMsg']),
