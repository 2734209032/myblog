# 3.条件渲染

Page({
    data:{
        type:1
    }
})

</view>

# 5.列表渲染

Page({
    data:{
        list:[{username:'zs',id:1},{username:'ls',id:2}]
    }
})
<view wx:for="{{list}}" wx:key="id">
  {{index}} - {{item.username}}
</view>
//默认数组的当前项的下标变量名默认为 index，数组当前项的变量名默认为 item 
//使用 wx:for-item 可以指定数组当前元素的变量名，
//使用 wx:for-index 可以指定数组当前下标的变量名：
<view wx:for="{{list}}" wx:for-index="idx" wx:for-item="itemName">
  {{idx}}: {{itemName.username}}
</view>

# 11.小程序生命周期函数

1.//应用的生命周期
//app.js
App({
  /**
   * 当小程序初始化完成时，会触发 onLaunch（全局只触发一次）
   */
  onLaunch: function () {
    console.log('onLaunch')
  },
  /**
   * 当小程序启动，或从后台进入前台显示，会触发 onShow ,就是用户从桌面点击了小程序，就会触发
   */
  onShow: function (options) {
    console.log('onShow')

  },
  /**
   * 当小程序从前台进入后台，会触发 onHide，用户从小程序回到桌面，触发
   */
  onHide: function () {
    console.log('onHide')

  },
  /**
   * 当小程序发生脚本错误，或者 api 调用失败时，会触发 onError 并带上错误信息
   */
  onError: function (msg) {
    console.log('onError')
  }
})
2.//页面的生命周期---------------------------------------------------------------------------------------------------
index.js
Page({
      /**
   * 生命周期函数--监听页面加载
   */
  2.1 onLoad: function (options) {  //相当于vue中的created
    console.log('index.onLoad')
  },
  /**
   * 生命周期函数--监听页面初次渲染完成
   */
 2.2 onReady: function () { //相当于vue中的mouted
    console.log('index.onReady')
  },
  /**
   * 生命周期函数--监听页面显示
   */
  2.3 onShow: function () {   就是用户从桌面点击了小程序，就会触发，tab栏切换回来也会触发
    console.log('index.onShow')
  },
  /**
   * 生命周期函数--监听页面隐藏
   */
  2.4 onHide: function () {   用户从小程序回到桌面，触发，tab栏切换也会触发
    console.log('index.onHide')
  },

  /**
   * 生命周期函数--监听页面卸载
   */
  onUnload: function () {   普通页面切换，不是像tab栏那样切换就会卸载
    console.log('index.onUnload')
  },
})

# 12.小程序发promise请求

//小程序是不支持promise的，但是我们用requset会造成回调嵌套，非常的麻烦，需要借用第三方插件
方式一-----------------------------------------------------
1.安装 npm i miniprogram-api-promise   构建npm工具
2.//在app.js封装全局的方法
import { promisifyAll } from 'miniprogram-api-promise';
const p = {}
promisifyAll(wx, p)
wx.p = p
3.在index.js
onLoad(){
    this.getList()
}
async getList(){
    const res = await .....
}
方式二------------------------------------------------------- 建议，支持原生小程序和uni-app
 1.安装npm i @escook/request-miniprogram    构建一下npm工具
 2.在app.js封装全局方法
// 按需导入 $http 对象
import { $http } from '@escook/request-miniprogram'
// 将按需导入的 $http 挂载到 wx 顶级对象之上，方便全局调用
wx.$http = $http
//设置基地址
$http.baseUrl = 'https://www.example.com'
3.发请求
// 发起 GET 请求，data 是可选的参数对象
$http.get(url, data?)
// 发起 POST 请求，data 是可选的参数对象
$http.post(url, data?)
// 发起 PUT 请求，data 是可选的参数对象
$http.put(url, data?)
// 发起 DELETE 请求，data 是可选的参数对象
$http.delete(url, data?)
 eag：
      onLoad(){
             this.getList()
         }
    async getList(){
        const res  = await wx.$http.get('url',data)
    }
4.它还支持请求拦截器  相关去npm官网查看

# 14.提示加载消息

1. // 提示加载中
    wx.showLoading({
      title: '加载中...',
    })
2.  //数据请求完成，取消加载的loading
      wx.hideLoading()

//请求防抖效果
 if (this.data.timer) {
      clearTimeout(this.data.timer)
    }
    // 提示加载中
    wx.showLoading({
      title: '加载中...',
    })
    let timer = setTimeout(async () => {

      const res = await wx.$http.get('/v1_0/articles',
        {
          timestamp: this.data.timestamp,
          channel_id: 0,
          with_top: 1
        })
      // console.log(res)
      this.setData({
        list: [...this.data.list, ...res.data.data.results],
        timestamp: res.data.data.pre_timestamp
      })
      //数据请求完成，取消加载的loading
      wx.hideLoading()
    }, 500)
    this.setData({ timer })
3.wx.showToast //提示成功
4.wx.showModal //弹出框 有确认和取消
5.wx.setStorageSync(Object object)  //存储
