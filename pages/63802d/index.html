# 3.路由模式

/ routr/index.js
// router  路由的全局最大的一个实例，唯一的
// routes  路由规则列表 
// route   一条路由规则 
Vue.use(VueRouter)
const router = new VueRouter({
    // 路由模式 两种
    // 1. hash模式 ==> # 
    // 2. history模式  ===> /不带#号 
    //  （如果改成了history模式，以后上下后端需要配置一些东西） 
    // 后端如果不会，把vue文档丢给他，来，我教你~~~ 

    mode:'history',  //设置路由模式 ，默认是hash
    routes:[
        // redirect后跟的是一个路径
        // 当一进入页面，匹配到/的时候，重定向到/login
        {path:'/', redirect:'/login'},
        // 简写成单个单词，只是方便我们使用，一般情况下多单词
        {path:'/login', component:Login},
        {path:'/article', component:Article},
        {path:'/list', component:List},
        {path:'/register', component:Register},
        // url地址上输入==>  /user/007
        // 如果要传多个参数，最好建议 先写一个自定义的名字再写动态key
        // /user/:id  :id可以动态的设置 
        {path:'/user/:id/color/:color', component:User},

        // 路由优先级：路由定义得越早，优先级就越高。
        // 含有通配符的路由应该放在最后
        // 404配置，一般在路由规则的最后一行配置
        {path:'*', component:NotFound}
    ]  
})

# 4.Object.defineProperty()

 Object.defineProperty()
        // 语法：Object.defineProperty(obj, prop/key, descriptor)
        // 作用：给对象定义一个属性或者修改原来的一个属性，并返回这个对象
       
        const obj = {
            id:1,
            name:'小米',
            price:1999
        }

        // 从对象上删除某个属性 ==>  delete obj.price
        // 想给obj这个对象，添加一个属性 color 颜色 ，怎么做呢 ？
        // obj.color = 'red'
        // console.log(obj)

        // 可枚举，遍历对象的属性的意思，
        Object.defineProperty(obj, 'color', {
            value:'red',  // 设置属性值,
            enumerable:true, // 控制属性是否可以枚举，默认为false
            configurable:true, // 主要控制对象的这个属性是否可以被删除，默认为false
            writable:true,  // 控制对象的当前这一个属性是否可以被修改（设置、赋值） = ， 默认false
        })
        console.log(obj)

        // 遍历一个对象的属性 
        for (el in obj){
            console.log(el)
        }
        const res = Object.keys(obj)
        console.log(res)

# 5.响应式getter和setter函数

 let color = 'orange'
        
        const obj = {
            id:1,
            name:'小米',
            price:1999
        }

        Object.defineProperty(obj, 'color', {
            enumerable:true, // 控制属性是否可以枚举，默认为false
            configurable:true, // 主要控制对象的这个属性是否可以被删除，默认为false

            // 存取描述符 一般叫这两个叫做 getter函数 和 setter函数 

            // 当读取obj.color的时候，getter函数就会被调用，返回值就是color属性的值
            get:function reactiveGetter(){
                console.log('getter函数被调用了')
                return color
            },
            // obj.color = 'skyblue'
            // 当属性值被修改的时候，会调用这个函数，它接收一个参数
            set:function reactiveSetter(newVal){
                console.log('setter函数被调用了')
                color = newVal
            }
        })

        // 注意：数据描述符 value 和 writable 不能和 get 或 set函数一起使用
        console.log(obj)
        // console.log(obj.color)

        // 通过get和set函数，让两个独立的obj 和 color变量 产生了一种微妙的联系
    </script>

# 6.数据代理

  // 数据代理： 通过一个对象代理对另一个对象的中属性的操作（读/写）

        // data(){
        //     return {
        //         msg:'hello world '
        //     }
        // }
        // this.msg ===> 

        // this.msg  ==>  this._data.msg ==> this.$data.msg ==> {
        //         msg:'hello world '
        //     }
        // this这个对象，代理了 data中return出来的对象
        // this._data这个对象，代理了 data中return出来的对象

        let obj = {}
        let obj2 = {x:100, y:200}

        // obj本身目前没有x属性，没有y属性 ，
        // 想要，访问obj.x  ==> obj2.x 
        //      访问obj.y  ==> obj2.y

                // obj.x = '某个值'  ==> obj2.x的值

        // obj.x ==>  自动调用getter函数
        Object.defineProperty(obj, 'x', {
            configurable:true,
            enumerable:true,
            //value + writable  ==> 不能 和  get 和set共存 
            get: function(){
                return obj2.x 
            },
            set(newVal){
                // 处理逻辑
                console.log('setter函数被调用了')
                // obj.x = '某个值'
                obj2.x = newVal
            }
        })

        // 我们现在让 obj 这个对象，代理了 obj2的x属性      
        // 让obj这个对象  ==> 代理 obj2这个对象

        console.log(this.msg)

    console.log(vm)

    // vm.msg  ===>  访问的是 data函数return出来的对象中的msg

  1. Vue中的数据代理
    // 通过vm对象代理了data对象中属性的操作（读/写）

    // 更加方便的操作data中的数据 
    // this.msg  =>  vm._data.msg

        // 处理逻辑
        console.log('setter函数被调用了')
        // obj.x = '某个值'
        data.age = newVal
      }
    })
        // vm.msg ==> data.msg

# 8.vue响应式原理

// 数据劫持  ==> 监听对象的所有属性 劫持了这个对象
    //             我们对这个对象的属性操作的时候，我都能知道~~~
    // obj.msg         ==> getter 
    // obj.msg = 'hi'  ==> setter 

    // 数据劫持+发布订阅

    // Vue是一个数据驱动的框架 
    // MVVM 
    // M : Model       数据模型 
    // V : View        视图
    // VM: viewModel   视图数据模型 ==> Vue实例vm

    // 当数据变化了 ==通过vm做了处理=> 更新视图

    // 响应式 ==> 数据变化了，Vue响应式的更新视图 

     data(){
         return {
             msg:'hi'
        }
     }

    // this.msg = 'hello'  ==> 视图更新 
    // 当我们去设置data中的属性值的时候，触发了setter函数 ，
    // setter函数里面就可以去做一些更新视图的操作 

    // 定义个函数，update，来表示做视图更新的操作
     let data = {
        msg:'hello',
         age: 18,
         // 如果对象里面嵌套对象，还会继续递归的监听对象的所有属性 
         demo:{
             name:{
            }
         }
     }
    const update = function () {
      // 内部做一些视图更新的逻辑~~~ 
      console.log('视图更新啦~~~')
    }

    // 我想要 data.msg变化了，就去更新视图 
    // data.msg  ==> 
    // 定义一个defineReactive 将data数据做监听，响应式处理 
    // 这个函数的作用就是对某个对象的属性做监听
    function defineReactive(obj, key, val) {
      Object.defineProperty(obj, key, {
        configurable: true,
        enumerable: true,
        get: function reactiveGetter() {
          // 获取了data中的值 
          console.log('触发了getter函数')
          return val
        },
        set: function reactiveSetter(newVal) {
          console.log('触发了setter函数')
          // 修改了data中的值 
          if (val === newVal) return
          // 更新视图 
          update(newVal) // vm.msg = 'hi' newVal ==> hi
        }
      })
    }
    // initData 
    function observer(value) {
      // {msg:'hell',age:18}
      if (!value || typeof value !== 'object') return

      // ['msg', 'age']
      // 实际上会做递归处理，
      Object.keys(value).forEach(key => {
        // value => data 
        // key   =>  msg 
        // value[key]  => I am hi 
        defineReactive(value, key, value[key])
      })
    }
    // 声明一个Vue类，构造函数 
    class Vue {
      // new Vue的时候 ，调用 constructor()
      constructor(options) {
        this._data = options.data
        // 对data中所有的属性做监听 
        observer(this._data)
      }
    }
    const vm = new Vue({
      data: {
        test: 'I am hi~~'
      }
    })

    // vm._data.test = 'hello world '

    console.log(vm)

# 9.vue2 对象响应式缺陷

就是我们创建项目的时候，在data函数return的对象，在一开始就被监听了，是响应式的，但我们在后来为这个对象添加属性，就没有被监听到，就是非响应式的
const vm = new Vue({
            el:'#app',
            data(){
                return {
                    obj:{
                        a:1
                    },
                }
            },
            mounted() {
                // Vue.set()
                // this.$set(obj,'c', 999)
            },
        })
        vm.b = 2 
        console.log(vm)

        // Object.defineProperty，或者Vue2的缺陷
        // 1. 针对对象  Vue 无法检测 property 的添加或移除

        // 2. 但是可以使用Vue.set或者this.$set给嵌套对象添加响应式属性

        // vm.a ==> proxy()
        // Vue.set(vm.data, '')
        
        // console.log(vm.obj)
        // 全局的Vue.set方法 （哪个对象，哪个属性，值）
        Vue.set(vm.obj, 'b', 2)
        console.log(vm)
        // 等效于  
        vm.$set(vm.obj, 'b', 66)
    </script>
