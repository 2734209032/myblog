        1. include 包含，列出来的组件才缓存

# 7.3 beforeRouteLeave

beforeRouteLeave  组件内守卫
//在组件里跟声明周期函数一样，和methods data props同级
 // beforeRouteLeave,
// 在某一个组件内部，输入很多表单信息， 如果不小心点到了其他路由，一下子要跳转了
// 我们可以在离开当前组件的路由之前，拦截 ，询问

 beforeRouteLeave(to, from, next) {
                // 在导航离开渲染该组件的对应路由时调用
                // 与 `beforeRouteUpdate` 一样，它可以访问组件实例 `this`
                if(window.confirm('你确定要离开吗？真的要离开吗？')){
                    next()
                } else {
                    next(false)
                }

            },

# 7.5全局守卫

 // 一 全局守卫
        // 1. 全局前置守卫 router.beforeEach((to,from,next)=>{})
        // 2. 全局解析守卫 router.beforeResolve()
        // 3. 全局后置守卫 router.afterEach((to,from) =>{})    ==>它唯一没有next函数
        router.beforeEach((to,from,next)=>{
            console.log('beforeEach')
            next()
        })

        router.beforeResolve((to,from,next)=>{
            console.log('beforeResolve')
            next()
        })

        router.afterEach((to, from) => {
            // to and from are both route objects.
            console.log('afterEach')
            // console.log(to, from)
        })


///----------------------------------------------------------------------------------------------
router.beforeEach 在渲染该组件的对应路由被验证前调用，准备跳转的时候调用
//在router/index
// to:Route  往哪儿去，即将进入的那个路由对象
// from:Route 从哪儿来，当前导航正要离开的路由对象
// next:Function  是否放行
//              1. next()  直接调用，不传参，不做处理，放行
//              2. next('/路径') 拦截到一个不同的地址去
//              3. 确保next函数，只调用一次，必须要调用一次
const whiteList = ['/login', '/register'] // 白名单列表，记录无需权限访问的所有页面   只是个变量，命名更语义化
// 导航守卫存在的意义：可以加拦截判断
// 1. 判断用户有没有token，如果有，直接放行
// 2. 没有token，白名单的页面（登录，注册），这些可以访问
// 3. 没有token，要去的还是首页等需要授权的页面，拦截到登录
router.beforeEach((to, from, next) => {
  const token = getToken()
  if (token) {
    next()
  } else {
    // 判断 一个字符串在数组中存在 includes / indexOf 判断是否是白名单的路径
    if (whiteList.includes(to.path)) {   //to.path ，to是个对象，to.path表示当前点击跳转的路由路径
      next() // 如果是白名单 则放行
    } else {
      // 不是白名单 也没有token 强行跳转到，登录页 去登录
      next('/login')
      Toast.fail('请登录!')
    }
  }
})
