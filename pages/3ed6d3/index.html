# 1.promise的.then的返回值

 // 1. 自己封装一个返回Promise函数
  function timeoutPro() {
    return new Promise((resolve, reject) => {
      // 执行异步
      setTimeout(() => {
        resolve('延迟1秒执行啦')
      }, 1000)
      // 成功 resolve-->then
    })
  }
  // .then里面不管有没有写返回值，它的执行结果都是一个promsie
  // 第一个then里面的回调函数，如果有返回一个promise,并且有resolve,那么第二个then里面的res的值就是第一个then里面的resolve的值
  // 第一个then里面的回调函数，如果返回的是一个具体的值，第二个then里面的res会接收到这一个具体的值
  // 第一个then里面的回调函数，如果没有返回值（任何值）,第二个then里面的res的结果是：undefined

  let p = timeoutPro()
    .then((res) => {
      //   return timeoutPro()
      return '你好吗'
    })
    .then((res) => {
      console.log(res)
    })
  console.dir(p)

# 2.promise中.then的第二个参数

 // 1. 随机数Promise
  function randomPro() {
    return new Promise((resolve, reject) => {
      // 延迟 1s
      setTimeout(() => {
        // 随机数
        // Math.random() 0-1的小数
        // * 100 控制范围
        // parseInt 转整数
        const num = parseInt(Math.random() * 100)
        if (num > 0 && num < 50) {
          resolve('成功啦:' + num)
        } else if (num >= 50 && num <= 100) {
          reject('失败啦:' + num)
        }
      }, 1000)
    })
  }
  // then的第二个参数会在失败的时候触发（只要你写了）
  // then的第二个参数就和catch都写了的时候，会优先触发第二个参数
  randomPro()
    .then(
      (res) => {
        console.log(res)                //.then其实是有第二个参数的，每个参数都是一个回调函数，第一个参数是，成功了执行的                                         回调函数，第二个参数是 失败了执行里面的回调函数，但以前，失败的时候我们用的catch接                                        收，现在当.then里面有第二个参数，就优先执行这个回调函数，不会执行catch里面的了

      },
      (err1) => {
        console.log('err1', err1)
      }
    )
    .catch((err) => {
      console.log(err)
    })

# 3.promise中.then中的第二个参数和catch的区别

<script>
    let p = new Promise((resolve, reject) => {
      resolve('22')
      // throw new Error('test')
    })
    // 如果是在promise内部抛出的错误，.then的第二个参数和.catch都可以捕获到他
    // 如果不是在promise内部抛出的错误，而是在外部（.then里面的第一个参数中），只有.catch可以捕获到它
    p.then(
      (res) => {
        throw new Error('test')
      },
      (err) => {
        console.log('err', err)
      }
    ).catch((err1) => {
      console.log('err1', err1)
    })
  </script>

# 4.promise的三种状态

 // 1. 随机数Promise
    function randomPro() {
      return new Promise((resolve, reject) => {
        // 延迟 1s
        setTimeout(() => {
          // 随机数
          // Math.random() 0-1的小数
          // * 100 控制范围
          // parseInt 转整数
          const num = parseInt(Math.random() * 100)
          if (num > 0 && num < 50) {
            resolve('成功啦:' + num)
          } else if (num >= 50 && num <= 100) {
            reject('失败啦:' + num)
          }
        }, 1000)
      })
    }

    let p = randomPro()
    console.log(p) // pending
    p.then(
      (res) => {
        console.log('res')
        console.log(p) // fulfilled
      },
      (err) => {
        console.log('err', err)
        console.log(p) // rejected
      }
    ).catch((err1) => {
      console.log('err1', err1)
    })

# 5.事件循环

老师的：
//     事件循环（eventloop）:
// 1、主线程上面的同步代码，
// 2、取出微任务列表里面所有的微任务，全部执行
// 3、取出宏任务队列里面的一个宏任务，执行
// 4、继续执行第2步
// 5、继续执行第3步
// 6、继续执行第2步
// 7、继续执行第3步
我的：
事件循环，最开始，执行script这个宏任务，执行过程中，同步任务立即执行，宏任务放入宏任务队列中，微任务放入微任务队列中，等整个script脚本这个宏任务执行完之后，再去微任务队列中看有没有需要执行的微任务如果有的话就依次执行，执行完了之后，去看宏任务队列中是否有宏任务，如果有拿一个出来执行，执行完了之后再去看微任务队列中有需要执行的微任务，如果有，全部执行完，再去宏任务里看，依次循环

# 8.async函数的返回值

script>
    function delay(s) {
      return new Promise((resolve, reject) => {
        setTimeout(() => {
          resolve(`延迟了${s}秒`)
        })
      })
    }
    async function fun1() {
      const res1 = await delay(1)
      console.log('res1', res1)
      const res2 = await delay(3)
      console.log('res2', res2)
    }
    async function fun2() {
      const res3 = await delay(2)
      console.log('res3', res3)
      const res4 = await delay(4)
      console.log('res4', res4)
    }
    //   fun1()
    //   fun2()
    // res1 res3 res2 res4
    // await函数只能影响async函数内部的执行顺序，并不能影响外部函数的执行顺序
    //   验证async函数的返回值是什么:promise
    //   let p = fun1()
    //   console.log('p', p) // promise

    // 想要先执行完fun1里面的所有代码，再去执行func2里面的代码，又该怎么做？
    async function yyy() {
      await fun1()
      await fun2()
    }
    yyy() // res1 res2 res3 res4

    // 以上代码等价于：
    fun1().then((res) => {
      fun2()
    })

    // 再次证明：
    //await函数只能影响async函数内部的执行顺序，并不能影响外部函数的执行顺序
    console.log('top111')
    yyy()
    console.log('bottom111')
    // top111 bottom111 res1 res2 res3 res4

  await async2()        // 注意： await 这一行是同步执行的

async function async2() {
  console.log('async2')
}

console.log('script start')

setTimeout(function () {
  console.log('setTimeout')  // 宏-1

async1()

new Promise(resolve => {
  console.log('Promise');
  resolve();
}).then(() => {
  console.log('promise1') // 微-2
}).then(() => {
  console.log('promise2') // 微-3

// 3. async2
