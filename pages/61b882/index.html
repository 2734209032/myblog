# 1.slot默认插槽和具名插槽

// 插槽的分类 
   1.默认插槽 不带有name属性的slot  => 默认名字就叫default
  <slot></slot> 等效于 ==> <slot name="default"></slot>
<!-- slot插槽，作用：提前占位，将来写在父组件标签中间的那部分内容，就会替换这个标签 -->
      <!-- slot标签中也可以写默认的内容，只会在父组件使用时，没有提供内容的时候被渲染 -->

  2.具名插槽 带有name属性的slot， 可以实现内容的定向分发
  2.1 带有插槽的组件中，在slot标签上写上name属性，来决定放哪块内容（双孔还是三孔的插座）
 <slot name="header"></slot>
   2.2 组件使用时，template + v-slot:name 来使用 
 <组件名>
    <template v-slot:header>
   <h3>温馨提示</h3>
 </template>
</组件名>

 3. 具名插槽  template + v-slot:name 使用的时候，可以简写
<组件名>
   <template #header>
   <h3>温馨提示</h3>
 </template>
</组件名>


// 注意：
 1. template标签不会影响任何结构内容，它只是一个包裹的容器，将来会被替换掉
 2. v-slot只能用于template标签 

//使用场景
  1. 简单使用场景，如果只是父组件传数据到子组件  ==> props可以实现
 2. 如果要定制化组件的内部某些结构 ==> 使用slot插槽 ，用于结构的内容分发

# 4.生命周期钩子函数

1. beforeCreate：data数据初始化之前，组件还没有数据
2. created: data数据初始化之后，可以获取到组件的数据
3. beforeMount：DOM挂载到页面之前，DOM还未挂载
4. mounted：DOM挂载之后，可以操作DOM了
5. beforeUpdate: 数据更新，DOM更新前  //组件更新前 --->数据是新的，页面是旧的 虚拟dom还没有对比重新渲染~~
6. updated: 数据更新，DOM更新后
7. beforeDestroy: 组件销毁前   // 组件销毁前 ——组件实例依然还保留全部的功能 // 全局的定时器，不会自动销毁，需要手动关闭~~
8. destroyed: 组件销毁后

// 初始化前 - 初始化数据和方法之前， => 此时无法访问组件中data的数据 和 方法 
  beforeCreate() {
    // console.log('beforeCreate', this.msg)
    console.log('-------beforeCreate---------')

    console.log(this.$data, '---', this.$el)
    console.log('----------------------------')
  },
  // 初始化后 重点， 此时，是Vue中最早可以访问data数据的生命周期中 => 可以访问数据和方法了
  // 但是 ==>  created生命周期里面，是不能访问到dom元素的！！！
  async created() {
    // console.log('created', this.msg)

    const res = await this.$http.get(api)
    this.arr = res.data.data
    console.log(this.arr)
  }

# 7.router

可以在main的js写 ，但是当数据多的时候就显得繁琐   ，所以我们一般在src下创建一个router文件夹，再在router下创一个index.js,
 在index.js下写我们的逻辑
1.下载router  npm i vue-router@3.6.5
2.导入router  import VueRouter from 'vue-router' //当然在main.js里面已经导入了Vue，我们这里需要导入，导入需要切换的组件
3.使用Vue.use(VueRouter) =>帮我们安装插件用的一个方法
4.创建router实例 const router = new VueRouter({})  //这里构造函数的参数是一个配置项，下面会使用
5.在main.js里面把router注入到vue根实例上
new Vue({
  render: h => h(App),
  router
}).$mount('#app')
//核心两大步骤
6.配置路由规则  //也就是在实例化router构造函数里面写
const router = new VueRouter({
  routes: [  //注意这里是routes
    { path: '/find', component: find },   //path是ip地址上#后面的地址 component是组件名
    { path: '/my', component: my },
    { path: '/part', component: part },
  ],
7.指定路由出口 挖坑占位，告诉我们组件应该在哪里渲染展示 //一般我们是在App.vue里面
    <router-view></router-view>   ///router-view本质上是vue-router提供给我们的一个全局组件

      <a href="#/part">朋友</a> -->
//<router-link>用法
      <router-link to="/find">发现音乐</router-link>   ///点击的时候自动给我们添加了两个类，可以给类添加样式，达到高亮效果
      <router-link to="/my">我的音乐</router-link>  ///本质上还是把这个转换为a标签，源码默认是a标签 可以修改

        router-link-active        模糊匹配           //只要开头是我们<router-link>里面的to的值，就能加上这个类

# 10.Vue.use源码

import type { GlobalAPI } from 'types/global-api'
import { toArray, isFunction } from '../util/index'

export function initUse(Vue: GlobalAPI) {
  // Vue.use是一个函数 接收一个插件作为参数
  Vue.use = function (plugin: Function | any) {
    // 声明一个数组来存安装过的插件~~~
    const installedPlugins =
      this._installedPlugins || (this._installedPlugins = [])
    // 如果找得到这个插件，说明插件安装过，return this
    if (installedPlugins.indexOf(plugin) > -1) {
      return this
    }

    // additional parameters
    const args = toArray(arguments, 1)
    // 添加Vue实例到参数中去
    args.unshift(this)
    // Vue的插件，按Vue的规范，应该要有一个install方法
    if (isFunction(plugin.install)) {
      // apply改变this指向  ==> 执行plugin内部的install方法
      // apply 会立即执行调用函数
      plugin.install.apply(plugin, args)
      // 如果插件本身是一个function
    } else if (isFunction(plugin)) {
      // 执行这个函数
      plugin.apply(null, args)
    }
    // 用数组存起来~~~
    installedPlugins.push(plugin)
    // return Vue实例
    return this
  }
}
// Vue.use 本身是一个函数，用来安装插件~~
// Vue.use内部做了哪些操作？
// 1. 判断插件是否注册安装过
// 2. 如果这个要安装的插件本身有install方法，执行它
//    同时把Vue构造函数作为第一个参数，以及use中的其他参数传入
// 3. 如果没有install方法，并且插件本身就是一个函数，就去调用执行它
//    同时把Vue构造函数作为第一个参数，以及use中的其他参数传入
// 4. 记录这个插件已经注册过
