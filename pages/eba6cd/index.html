# 1.微信一键登录

<view class="btn-box">
  <van-button type="primary" block wx:if="{{!hasLogin}}" bindtap="onLogin">一键登录</van-button>
  <van-button type="danger" block wx:else bindtap="onLogout">退出</van-button>
</view>

let globalToken = getApp().globalData.token
Page({
  data: {
    hasLogin: globalToken
  },
  onLoad(options) {
    console.log('Login onLoad')
    this.setData({
      hasLogin: globalToken
    })
  },
  async onLogin() {
    // 1.获取code，是用户临时的标识
    let { code } = await wx.login()
    // 2.获取用户信息，经过加密的信息(包含手机号，性别...)x需要后端进行解密
    let { encryptedData, rawData, iv, signature } = await wx.getUserInfo()
    // console.log(res)
    //3.发请求
    let { data: { message: { token } } } = await wx.$http.post('/users/wxlogin', {
      code, encryptedData, rawData, iv, signature
    })
    // console.log(token)
    // 4.存储token
    wx.setStorageSync('token', token)
    globalToken = token//在根组件存储token
    this.setData({
      hasLogin: true
    })
  },
  onLogout() {
    wx.removeStorageSync('token')
    globalToken = undefined
    this.setData({
      hasLogin: false
    })
  },

})

# 2.微信支付

<van-submit-bar price="{{ total*10 }}" button-text="提交订单" bind:submit="onSubmit" />
1
Page({
  data: {
    goodsList: [{ goods_id: 1, goods_name: '测试商品', goods_price: 0.1, goods_number: 1 }],
    total: 0.1,
    token: undefined
  },
  onLoad(options) {
    let globalToken = getApp().globalData.token
    this.setData({
      token: globalToken
    })
  },
  async onSubmit() {
    try {
      //1.判断用户是否登录
      // console.log(token)
      if (!this.data.token) {
        wx.navigateTo({
          url: '/pages/login/login',
        })
        return wx.showToast({
          title: '请先登录',
          icon: 'error'
        })
      }
      //2.创建订单 得到订单号 
      let { data: { message: { order_number } } } = await wx.$http.post('/my/orders/create', {
        order_price: this.data.total,
        consignee_addr: '四川省成都市金牛区xxx小区',
        goods: this.data.goodsList
      })
      console.log(order_number)
      //3.预支付，获取调用支付的数据
      let { data: { message: { pay } } } = await wx.$http.post('/my/orders/req_unifiedorder', { order_number })
      //4.调用微信支付api，弹出支付弹层
      await wx.requestPayment(pay)
      //5.查询订单是否支付完成
      // 等2-3秒
      wx.showLoading({
        title: '支付中...'
      })
      await new Promise((resolve, reject) => {
        setTimeout(() => {
          resolve()
        }, 2000)
      })
      //发请求 
      let { data: { meta } } = await wx.$http.post('/my/orders/chkOrder', { order_number })
      //6.判断是否支付成功
      if (meta.status === 200) {
        wx.showToast({
          title: '支付成功',
          icon: 'success'
        })
      }
      else {
        wx.showToast({
          title: '支付失败',
          icon: 'error'
        })
      }
    } catch (err) {
      wx.showToast({
        title: '支付失败',
        icon: 'error'
      })
    }
